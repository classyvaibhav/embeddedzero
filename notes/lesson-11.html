1The Project: Environmental Sensor Hub

You'll build a device that reads temperature, humidity, and pressure, displays them on an OLED screen, and lets you toggle between screens with a button. This combines everything you've learned: GPIO, I2C, button debouncing, display driving.

📊 What You're Building A pocket-sized weather station. Reads DHT22 (temp + humidity) + BMP280 (pressure + altitude). Displays on a 128×64 OLED. Press a button to cycle views: Summary / Detailed / Historical Graph.

3Connection Guide

Wire each ESP32 pin to the corresponding pin on each module. This table is how professional hardware engineers actually document wiring — far easier to follow than a schematic with crossing wires.

ESP32 PinGoes ToPurpose
3V3OLED V+, BMP280 V+, DHT22 V+Power (3.3 V) for all 3 sensors
GNDOLED GND, BMP280 GND, DHT22 GND, Button (one leg)Common ground for everything
GPIO 21 (SDA)OLED SDA and BMP280 SDAI2C data — both sensors share this wire
GPIO 22 (SCL)OLED SCL and BMP280 SCLI2C clock — both sensors share this wire
GPIO 5DHT22 DATASensor data (single-wire protocol)
GPIO 4Button (other leg)Screen toggle input (INPUT_PULLUP)
⚠ Don't Forget
  • Add a 10 kΩ pull-up resistor between the DHT22 DATA pin and 3V3. DHT22 won't communicate without it.
  • OLED and BMP280 are both I2C — they share SDA/SCL wires but have different addresses (0x3C and 0x76). The library handles address selection automatically.
  • Button uses INPUT_PULLUP in code, so no external pull-up needed for it.

How the I2C Bus Works

The clever part of this circuit is how OLED and BMP280 both connect to the same two wires:

I2C bus — one master (ESP32), two slaves sharing two wires.
💡 Easier Alternative Many hobbyists use a small 4-pin or 6-pin terminal strip on a breadboard to distribute 3V3 and GND across multiple modules. It keeps the wiring cleaner. Once you add a third I2C device, this becomes essential.

4The Complete Code

#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <Adafruit_BMP280.h>
#include <DHT.h>

#define DHT_PIN 5
#define BTN_PIN 4
#define DHT_TYPE DHT22

Adafruit_SSD1306 display(128, 64);
Adafruit_BMP280 bmp;
DHT dht(DHT_PIN, DHT_TYPE);

int screen = 0;
volatile bool btnEvent = false;
unsigned long lastBtn = 0;
unsigned long lastRead = 0;
float temp = 0, hum = 0, press = 0;

void IRAM_ATTR onButton() {
  unsigned long now = millis();
  if (now - lastBtn > 200) {
    btnEvent = true;
    lastBtn = now;
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(BTN_PIN, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(BTN_PIN), onButton, FALLING);

  Wire.begin();
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  bmp.begin(0x76);
  dht.begin();

  display.setTextColor(WHITE);
  display.setTextSize(1);
  display.clearDisplay();
  display.setCursor(0, 0);
  display.println("Sensor Hub");
  display.println("Bahar's Project");
  display.display();
  delay(1500);
}

void loop() {
  unsigned long now = millis();

  // Read sensors every 2 seconds
  if (now - lastRead >= 2000) {
    lastRead = now;
    temp = dht.readTemperature();
    hum = dht.readHumidity();
    press = bmp.readPressure() / 100.0F;  // convert Pa to hPa
  }

  // Handle button: cycle screens
  if (btnEvent) {
    btnEvent = false;
    screen = (screen + 1) % 3;
  }

  // Render current screen
  display.clearDisplay();
  display.setCursor(0, 0);

  if (screen == 0) {
    display.println("SUMMARY");
    display.println();
    display.print("T: "); display.print(temp, 1); display.println(" C");
    display.print("H: "); display.print(hum, 1); display.println(" %");
    display.print("P: "); display.print(press, 0); display.println(" hPa");
  } else if (screen == 1) {
    display.println("DETAILED");
    display.println();
    display.print("Temp: "); display.print(temp, 2); display.println(" C");
    display.print("Humid: "); display.print(hum, 1); display.println(" %");
    display.print("Press: "); display.print(press, 2);
    display.println(" hPa");
    display.print("Alt: "); display.print(bmp.readAltitude(1013.25), 1);
    display.println(" m");
  } else {
    display.println("ABOUT");
    display.println();
    display.println("Bahar's Weather");
    display.println("Station v1.0");
    display.println("ESP32+DHT22+BMP");
  }
  display.display();
  delay(100);
}
